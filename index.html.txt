    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const trackingForm = document.getElementById('trackingForm');
            const trackingNumberInput = document.getElementById('trackingNumber');
            const resultSection = document.getElementById('resultSection');
            const loadingElement = document.getElementById('loading');
            const errorMessage = document.getElementById('errorMessage');
            const courierLogos = document.querySelectorAll('.courier-logo');
            let selectedCourier = 'jne';
            
            // Set default courier
            courierLogos.forEach(logo => {
                if (logo.classList.contains('active')) {
                    selectedCourier = logo.getAttribute('data-courier');
                }
            });
            
            // Courier selection
            courierLogos.forEach(logo => {
                logo.addEventListener('click', function() {
                    courierLogos.forEach(l => l.classList.remove('active'));
                    this.classList.add('active');
                    selectedCourier = this.getAttribute('data-courier');
                });
            });
            
            // Form submission
            trackingForm.addEventListener('submit', function(e) {
                e.preventDefault();
                
                const trackingNumber = trackingNumberInput.value.trim();
                
                if (!trackingNumber) {
                    showError();
                    return;
                }
                
                // Show loading
                loadingElement.classList.add('active');
                resultSection.classList.remove('active');
                errorMessage.classList.remove('active');
                
                // Simulate API call with timeout
                setTimeout(() => {
                    loadingElement.classList.remove('active');
                    
                    if (isValidTrackingNumber(trackingNumber)) {
                        displayTrackingResult(trackingNumber, selectedCourier);
                    } else {
                        showError();
                    }
                }, 1500);
            });
            
            function isValidTrackingNumber(trackingNumber) {
                // Simple validation - in a real app, this would call an API
                return trackingNumber.length >= 8 && trackingNumber.length <= 20;
            }
            
            function showError() {
                errorMessage.classList.add('active');
                resultSection.classList.remove('active');
            }
            
            function displayTrackingResult(trackingNumber, courier) {
                // Update result section with tracking data
                document.getElementById('trackingNumberResult').textContent = `Resi: ${trackingNumber}`;
                
                // Set random status for demo
                const statuses = [
                    { text: 'Terkirim', class: 'status-delivered' },
                    { text: 'Dalam Perjalanan', class: 'status-in-transit' },
                    { text: 'Menunggu Pickup', class: 'status-pending' },
                    { text: 'Ada Masalah', class: 'status-problem' }
                ];
                
                const randomStatus = statuses[Math.floor(Math.random() * statuses.length)];
                document.getElementById('packageStatus').textContent = `Status: ${randomStatus.text}`;
                document.getElementById('statusBadge').textContent = randomStatus.text;
                document.getElementById('statusBadge').className = `status-badge ${randomStatus.class}`;
                
                // Set sender and receiver info
                document.getElementById('senderInfo').textContent = 'PT. Example Sender';
                document.getElementById('receiverInfo').textContent = 'Budi Santoso';
                document.getElementById('serviceType').textContent = courier.toUpperCase();
                document.getElementById('packageWeight').textContent = `${(Math.random() * 5 + 0.5).toFixed(1)} kg`;
                
                // Generate timeline
                const timeline = document.getElementById('timeline');
                timeline.innerHTML = '';
                
                const events = generateTimelineEvents(courier, randomStatus.text);
                
                events.forEach((event, index) => {
                    const timelineItem = document.createElement('div');
                    timelineItem.className = `timeline-item ${index === 0 ? 'active' : ''}`;
                    
                    timelineItem.innerHTML = `
                        <div class="timeline-date">${event.date}</div>
                        <div class="timeline-content">${event.description}</div>
                    `;
                    
                    timeline.appendChild(timelineItem);
                });
                
                // Show result section
                resultSection.classList.add('active');
            }
            
            function generateTimelineEvents(courier, status) {
                const baseEvents = [
                    { date: '25 Nov 2023, 08:30', description: 'Paket telah diterima di gudang sortir' },
                    { date: '24 Nov 2023, 14:15', description: 'Paket sedang dalam perjalanan ke kota tujuan' },
                    { date: '23 Nov 2023, 10:45', description: 'Paket telah dipickup oleh kurir' },
                    { date: '22 Nov 2023, 16:20', description: 'Pesanan telah diproses oleh penjual' }
                ];
                
                if (status === 'Terkirim') {
                    baseEvents.unshift(
                        { date: '26 Nov 2023, 11:30', description: 'Paket telah diterima oleh penerima' }
                    );
                }
                
                if (status === 'Ada Masalah') {
                    baseEvents.unshift(
                        { date: '26 Nov 2023, 09:15', description: 'Ada kendala dalam pengiriman paket' }
                    );
                }
                
                return baseEvents;
            }
            
            // Add some sample tracking numbers for demo
            trackingNumberInput.placeholder = `Contoh: ${generateSampleTrackingNumber(selectedCourier)}`;
            
            courierLogos.forEach(logo => {
                logo.addEventListener('click', function() {
                    selectedCourier = this.getAttribute('data-courier');
                    trackingNumberInput.placeholder = `Contoh: ${generateSampleTrackingNumber(selectedCourier)}`;
                });
            });
            
            function generateSampleTrackingNumber(courier) {
                const samples = {
                    jne: '1234567890',
                    tiki: 'TKI987654321',
                    pos: 'CP123456789ID',
                    jnt: 'JNT1234567890',
                    sicepat: 'SICEPAT1234567',
                    ninja: 'NINJA123456789'
                };
                
                return samples[courier] || '1234567890';
            }
        });
